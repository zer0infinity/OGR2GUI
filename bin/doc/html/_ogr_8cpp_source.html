41 OsType checkOS() {
42 #ifndef Q_OS_WIN32
43  return Other;
44 #else
45  if (QSysInfo::WordSize == 64)
46  return Win_64;
47  BOOL isWOW64 = FALSE;
48  LPFN_ISWOW64PROCESS fnIsWow64Process;

51  return Win_32;

53  return Error;
54  return isWOW64? WOW64 : Win_32;

65 bool Ogr::OpenOgr2ogr(QString command, QPushButton *btnExecute) {
66  bool resVal = false;
67  if(checkOS() == Win_64) {
68  ogr2ogr = new Ogr2ogrThread(command, btnExecute);
69  ogr2ogr->start();
70  resVal = ogr2ogr->isRunning();
71  } else if(checkOS() == WOW64) {
72  QString logPath = QDir::toNativeSeparators(QCoreApplication::applicationDirPath() + QDir::separator() + "ogr2ogr.log");
73  btnExecute->setEnabled(false);
74  process = new QProcess();
75  process->setProcessChannelMode(QProcess::MergedChannels);
76  process->setStandardOutputFile(logPath);
77  process->start(command);
78  process->waitForStarted();
79  resVal = process->waitForFinished();
80  btnExecute->setEnabled(true);
81  } else if(checkOS() == Win_32){
82  std::string cmd = command.toStdString();
83  std::wstring widestring = std::wstring(cmd.begin(), cmd.end());
84  LPWSTR lpwstr = const_cast<LPWSTR>(widestring.c_str());
85  STARTUPINFO si;
86  PROCESS_INFORMATION pi;
87  ZeroMemory(&si, sizeof(si));
88  si.cb = sizeof(si);
89  ZeroMemory(&pi, sizeof(pi));
90  resVal = CreateProcess(NULL, lpwstr, 0, 0, FALSE, CREATE_NO_WINDOW, 0, 0, &si, &pi);
91  WaitForSingleObject(pi.hProcess, INFINITE);
92  CloseHandle(pi.hProcess);
93  CloseHandle(pi.hThread);
94  }
95  return resVal;
96 }
